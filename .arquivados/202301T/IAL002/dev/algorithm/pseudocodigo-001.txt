    DECLARE N1, N2, D NUMÉRICO
    ESCREVA "Digite dois números"
    LEIA N1, N2
    SE N2 = 0 ENTÃO
        ESCREVA "Impossível dividir"
    SENÃO INÍCIO
        D = N1/N2
        ESCREVA "Divisão = ", D
    FIM

    DECLARE N1, N2, SN, MF NUMÉRICO
    ESCREVA "Informe as duas notas do Aluno: "
    LEIA N1, N2
    SN = N1 + N2
    MF = SN / 2
    SE MF >= 6 ENTÃO
        ESCREVA "Aluno Aprovado. Média final é ", MF
    SENÃO
        ESCREVA "Aluno Reprovado. Média final é ", MF 
FIM_ALGORITMO.
